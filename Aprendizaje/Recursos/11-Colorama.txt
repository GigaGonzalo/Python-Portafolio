def barra_progreso_simple(progreso, total, ancho=40):
    """Crea una barra de progreso simple"""
    porcentaje = progreso / total
    barras_completas = int(ancho * porcentaje)
    barras_faltantes = ancho - barras_completas
    
    # Determinar color segÃºn progreso
    if porcentaje < 0.3:
        color = Fore.RED
    elif porcentaje < 0.7:
        color = Fore.YELLOW
    else:
        color = Fore.GREEN
    
    barra = f"{color}{'â–ˆ' * barras_completas}{Fore.WHITE}{'â–‘' * barras_faltantes}"
    return f"{barra} {porcentaje:.1%}"

def barra_progreso_animada():
    """Muestra una barra de progreso animada"""
    total = 100
    print(f"\n{Fore.CYAN}ğŸ”„ PROCESANDO ARCHIVOS...{Style.RESET_ALL}")
    
    for i in range(total + 1):
        time.sleep(0.05)  # Simular trabajo
        barra = barra_progreso_simple(i, total)
        sys.stdout.write(f"\r{barra}")
        sys.stdout.flush()
    
    print(f"\n{Fore.GREEN}âœ… PROCESO COMPLETADO!{Style.RESET_ALL}")
